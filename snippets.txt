@PostMapping("/subir")
    public String escribirArchivos(@RequestParam("archivos") MultipartFile[] files) throws IOException {
        FileInputStream fis = null;
        FileOutputStream fos = null;
        File f = null;
        for (MultipartFile file : files) {
            try (InputStream fileContent = file.getInputStream()) {
                f = File.createTempFile("pdfTemp", null);
                fos = new FileOutputStream(f);
                int dato;
                while ((dato = fileContent.read()) != -1) {
                    fos.write(dato);
                }
                fos.close();

                //Crear archivo desde un temp
                FileInputStream fs = new FileInputStream(f);
                FileOutputStream os = new FileOutputStream("C:\\Peajes\\" + file.getOriginalFilename());
                int b;
                while ((b = fs.read()) != -1) {
                    os.write(b);
                }
                //CreacionTerminada
                fs = new FileInputStream("C:\\Peajes\\" + file.getOriginalFilename());
